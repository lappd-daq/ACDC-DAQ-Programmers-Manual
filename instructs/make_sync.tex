\begin{lstlisting}[style=c++]
void SuMo::make_sync()
{
	usb.createHandles();
	const unsigned int hi_cmd = 0x000B0010;
	unsigned int send_word = hi_cmd;
	usb.sendData(send_word);
	usb.freeHandles();
}
\end{lstlisting}
